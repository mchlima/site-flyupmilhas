```typescript
// app/composables/useScroll.ts
export function useScroll() {
  function scrollTo(id: string) {
    if (import.meta.client) {
      const el = document.getElementById(id)
      el?.scrollIntoView({ behavior: 'smooth', block: 'start' })
    }
  }
  return { scrollTo }
}
```

The `import.meta.client` guard is MANDATORY — without it, `document` access during SSR throws a ReferenceError and breaks the build. Do not use `process.client` (Nuxt 3 pattern, deprecated in Nuxt 4).
